\subsection{Evaluation Protocols}
\label{sec:meth-eval-protocols}

\textit{Define three evaluation settings used throughout the thesis:}
\begin{enumerate}
    \item \textbf{All-city (standard)}: Train and test on the full NAVSIM split (all cities mixed). Reproduces published numbers.
    \item \textbf{Single-city (zero-shot transfer)}: Train on city A, test on all cities. Produces a $4 \times 4$ generalization matrix.
    \item \textbf{Leave-one-city-out (LOCO)}: Train on 3 cities, test on the held-out city. More realistic multi-city regime.
\end{enumerate}

\textit{Define the generalization gap metric:}
\begin{equation}
    \Delta_\text{gen} = \text{PDMS}_\text{in-dist} - \text{PDMS}_\text{OOD}
\end{equation}

\textit{And the transfer ratio (from IROS):}
\begin{equation}
    R_\text{transfer} = \frac{\text{PDMS}_\text{OOD}}{\text{PDMS}_\text{in-dist}}
\end{equation}

\section{SSL Pre-Training Protocol}
\label{sec:meth-ssl-pretraining}

\textit{Describe the self-supervised pre-training setup for I-JEPA on nuScenes driving data (used for some DiffusionDrive experiments):}

\begin{itemize}
    \item \textbf{Architecture}: I-JEPA ViT-S/14
    \item \textbf{Pre-training data}: nuScenes training split (driving images)
    \item \textbf{Training}: 100 epochs, AdamW optimizer, cosine LR schedule
    \item \textbf{Hardware}: 2$\times$ H200 GPUs
    \item \textbf{Input resolutions}: Both rectangular (native aspect ratio) and square (center-crop) variants
\end{itemize}

\textit{For ImageNet-pretrained backbones (I-JEPA ViT-H/14, DINOv2 ViT-L/14, MAE ViT-L/16), we use the official released checkpoints without further pre-training.}

\section{Planning Architecture I: LAW with Swappable Backbones}
\label{sec:meth-law}

\textit{LAW (Latent World Model, ICLR 2025) serves as the primary test bed for the cross-city SSL study (RQ1--RQ3). The backbone is the only component we swap; everything else remains fixed for fair comparison.}

\subsection{Architecture Overview}
\label{sec:meth-law-arch}

\textit{Key components:}
\begin{itemize}
    \item Per-view spatial decoders (one per camera view)
    \item LID (Lift, Image, Depth) depth encoding for 3D-aware features
    \item Delta waypoint prediction (predicts offsets from previous waypoint)
    \item TransformerDecoder latent world model (auxiliary training loss)
\end{itemize}

\textit{The world model loss encourages the backbone to produce features useful for predicting future states, which may independently improve generalization.}

\subsection{Training Configuration}
\label{sec:meth-law-training}

\textit{Adam optimizer (not AdamW), lr=1e-4, no weight decay, no scheduler, 20 epochs, batch\_size=32. Backbone lr\_mult=0.1 when fine-tuning. No gradient clipping. Mixed precision (fp16). These match the official LAW paper's NAVSIM configuration (Section 5.2).}

\subsection{TransFuser Baseline}
\label{sec:meth-transfuser}

\textit{TransFuser: ResNet-34 camera + ResNet-34 LiDAR + GPT-style auto-regressive fusion + MLP waypoint regression. Simpler than LAW (no world model loss). Used to test whether cross-city findings are architecture-dependent or backbone-dependent.}

\section{Planning Architecture II: DiffusionDrive + SSL}
\label{sec:meth-diffusiondrive}

\textit{DiffusionDrive (CVPR 2025) is the base for our generative planning experiments. We integrate SSL backbones and create a camera-only variant.}

\subsection{DiffusionDrive Architecture}
\label{sec:meth-dd-arch}

\textit{Key components:}
\begin{enumerate}
    \item \textbf{Anchor trajectories}: K-means clustering on training set trajectories produces 256 prototypical trajectories (\texttt{trajectory\_anchors\_256.npy}). These serve as the denoising starting point instead of Gaussian noise.
    \item \textbf{Truncated diffusion}: Only 2 denoising steps at inference (vs.\ 50--1000 in standard diffusion). Starts from nearest anchor trajectory, not pure noise. Enables 45+ FPS.
    \item \textbf{Multi-scale cross-attention decoder}: Conditions denoising on visual features at multiple spatial scales.
    \item \textbf{Original backbone}: ResNet-34 camera + ResNet-34 LiDAR (TransFuser-style fusion).
\end{enumerate}

\subsection{SSL Backbone Integration}
\label{sec:meth-dd-ssl}

\textit{Modifications for SSL backbone integration:}
\begin{enumerate}
    \item \textbf{Replace ResNet-34 image encoder} with I-JEPA ViT-S/14 (or MAE ViT-L/16). ViT outputs $(B, N, C)$ patch tokens.
    \item \textbf{Feature adapter}: Reshape ViT tokens to spatial feature maps $(B, C, H, W)$ via learned linear projection + reshape. A lightweight FPN (Feature Pyramid Network) generates multi-scale features compatible with DiffusionDrive's \texttt{MultiScaleCrossAttention}.
    \item \textbf{Two resolution modes}: Rectangular (native aspect, preserving horizontal FOV) and square (224$\times$224 center-crop, matching ImageNet pre-training). Results show resolution choice matters ($\sim$1pp impact).
    \item \textbf{Frozen vs.\ trainable}: Both modes tested. Frozen backbone + trainable adapter provides best balance.
\end{enumerate}

\subsection{Camera-Only Variant}
\label{sec:meth-dd-camera-only}

\textit{Remove all LiDAR dependencies (\texttt{lidar\_pc}, \texttt{bev\_raster}). The model operates purely on multi-camera images + ego state. The LiDAR branch of TransFuserBackbone is replaced with a learned latent vector (A13 configuration). This targets edge deployment where LiDAR is unavailable or too expensive.}

\subsection{Training Configuration}
\label{sec:meth-dd-training}

\textit{AdamW optimizer, lr=1e-4, weight decay=1e-4, cosine LR with 3-epoch warmup, 100 epochs (reduced to 20 for some experiments). Batch size 128. DDP across 4 GPUs.}

\section{Experimental Protocol}
\label{sec:meth-exp-protocol}

\textit{Experiments are organized into 4 phases, each building on the previous:}

\subsection{Phase 1: Cross-City SSL Benchmark}
\label{sec:meth-phase1}

\textit{Uses LAW and TransFuser with all backbone configurations. 19 backbone configs evaluated on nuScenes per-city splits. Single-city training + $4 \times 4$ evaluation matrix + transfer ratio analysis. Addresses RQ1, RQ2, RQ3. (IROS 2026 submission.)}

\subsection{Phase 2: DiffusionDrive All-City}
\label{sec:meth-phase2}

\textit{DiffusionDrive with SSL backbones on standard NAVSIM (all cities). Establishes in-distribution performance before cross-city evaluation. Tests camera-only feasibility. 9 configurations (Table~\ref{tab:diffusiondrive-allcity}).}

\subsection{Phase 3: DiffusionDrive Cross-City}
\label{sec:meth-phase3}

\textit{Single-city training with DiffusionDrive + I-JEPA vs.\ ResNet34. 4 cities $\times$ 2 backbones = 8 training runs, each evaluated on all 4 city test sets = 32 evaluations. Tests whether Phase 1 findings extend to generative planners. Addresses RQ5.}

\subsection{Phase 4: Mechanistic Analysis}
\label{sec:meth-phase4}

\textit{``Why does SSL work?'' investigation using feature analysis (CKA, linear probing, attention maps, MMD, effective rank). No additional training required --- uses checkpoints from Phases 1--3. See Section~\ref{sec:exp-why-ssl} for results.}

\section{Metrics}
\label{sec:meth-metrics}

\textit{For every experiment, we report:}
\begin{itemize}
    \item \textbf{PDMS} (overall) and sub-scores: NC, DAC, EP, TTC, Comfort
    \item \textbf{EPDMS} (extended evaluation, where applicable --- DiffusionDrive models)
    \item \textbf{Generalization gap} $\Delta_\text{gen}$ and \textbf{transfer ratio} $R_\text{transfer}$
    \item \textbf{Inference speed} (FPS) for DiffusionDrive variants
    \item All results tracked via Weights \& Biases
\end{itemize}
